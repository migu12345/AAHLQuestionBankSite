```json
  "paper": "May 2024 Paper 2 TZ2",
  "session": "May 2024",
  "paper_type": "Paper 2",
  "question_number": "7(a)",
  "title": "Complex numbers argument",
  "topic": "Number and Algebra",
  "subtopic": "Complex numbers",
  "question_text": "...",
  "answer_text": "...",
  "marks": 4,
  "source": {
    "paper_file": "may_2024_p2_tz2.pdf",
    "markscheme_file": "may_2024_ms_p2_tz2.pdf",
    "paper_page": 6,
    "markscheme_page": 11

}
```
